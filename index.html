Integration Code for Your TAU Africa App

Add this code to your registration success handlers:

// After successful payment in your TAU Africa app function syncToDatabase(memberData) { // Method 1: Direct localStorage sync (same domain) const dbKey = 'tauAfricaDB'; const existingMembers = JSON.parse( localStorage.getItem(dbKey) || '[]' ); // Add member with unique ID const newMember = { id: generateMemberID(memberData.team), ...memberData, dateJoined: new Date().toISOString().split('T')[0], status: 'Active' }; existingMembers.push(newMember); localStorage.setItem(dbKey, JSON.stringify(existingMembers)); // Trigger notification sendWelcomeEmail(newMember); return newMember.id; } // Generate Member ID function function generateMemberID(team) { const prefix = team === 'TASK' ? 'TSK' : 'TLK'; const timestamp = Date.now().toString().slice(-6); const random = Math.floor(Math.random() * 1000) .toString().padStart(3, '0'); return `${prefix}-${timestamp}-${random}`; }
